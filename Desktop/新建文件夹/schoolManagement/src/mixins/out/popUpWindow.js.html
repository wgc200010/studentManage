export default {
    methods:{
        /**
         * 显示提交成功的弹窗
         */
        successPopup(){
                this.$message({
                type: 'success',
                message: '提交成功!'
            });
        },
        /**
         * 取消的弹窗
         */
        failPopup(){
            this.$message({
                type: 'info',
                message: '操作取消!'
            });
        }

    }
}
